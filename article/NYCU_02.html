func checkLength(man [][]int, woman [][]int) error {
  if len(man) != len(woman) {
    return fmt.Errorf("男女配對不正確")
  }

  if len(man) == 0 {
    return fmt.Errorf("清單為空")
  }

  count := len(man[0])

  for i := 0; i < len(man); i++ {
    if len(man[i]) != count {
      return fmt.Errorf("第 %d 個男生名單數量錯誤", i+1)
    }
    if len(woman[i]) != count {
      return fmt.Errorf("第 %d 個女生名單數量錯誤", i+1)
    }
  }

  return nil
}
